🦍 Monkey Climb — Stage 1
Climbing locomotion test. In VR: squeeze the grip button near a tree trunk, wall, or bar to grab it, then move your hand to pull your whole body around — let go to fling yourself with momentum, Gorilla-Tag style. On desktop: WASD + mouse to fly around and preview the map.

sun.position.set(30, 50, 20); sun.castShadow = true; sun.shadow.mapSize.set(2048, 2048); sun.shadow.camera.left = -60; sun.shadow.camera.right = 60; sun.shadow.camera.top = 60; sun.shadow.camera.bottom = -60; ctrl.add(handMesh); ctrl.addEventListener('squeezestart', () => onGripStart(i)); ctrl.addEventListener('squeezeend', () => onGripEnd(i)); // some Quest browsers fire 'select' for hand-tracking pinch too ctrl.addEventListener('selectstart', () => onGripStart(i)); scene.add(ground); } function makeClimbable(mesh, radiusOverride) { mesh.castShadow = true; mesh.receiveShadow = true; scene.add(mesh); const box = new THREE.Box3().setFromObject(mesh); const center = box.getCenter(new THREE.Vector3()); const size = box.getSize(new THREE.Vector3()); const radius = radiusOverride || (Math.max(size.x, size.y, size.z) / 2 + 0.15); climbables.push({ mesh, center, radius }); return mesh; } function buildForest() { // Decorative background trees (instanced, NOT climbable) for the "huge forest" feel const trunkGeo = new THREE.CylinderGeometry(0.35, 0.5, 5, 7); const leafGeo = new THREE.ConeGeometry(2.2, 5, 8); const trunkMat = new THREE.MeshStandardMaterial({ color: 0x5b4230, roughness: 1 }); const leafMat = new THREE.MeshStandardMaterial({ color: 0x2e6b3a, roughness: 0.9 }); const count = 260; const trunkMesh = new THREE.InstancedMesh(trunkGeo, trunkMat, count); const leafMesh = new THREE.InstancedMesh(leafGeo, leafMat, count); trunkMesh.castShadow = true; leafMesh.castShadow = true; const dummy = new THREE.Object3D(); let placed = 0; while (placed < count) { const x = (Math.random() - 0.5) * 300; const z = (Math.random() - 0.5) * 300; if (Math.sqrt(x * x + z * z) < 22) continue; // keep spawn area clear const s = 0.7 + Math.random() * 0.9; dummy.position.set(x, 2.5 * s, z); dummy.scale.set(s, s, s); dummy.rotation.y = Math.random() * Math.PI * 2; dummy.updateMatrix(); trunkMesh.setMatrixAt(placed, dummy.matrix); } function buildClimbingStructures() { const stoneMat = new THREE.MeshStandardMaterial({ color: 0x8a8a83, roughness: 1 }); const woodMat = new THREE.MeshStandardMaterial({ color: 0x9c6b3f, roughness: 0.9 }); const barMat = new THREE.MeshStandardMaterial({ color: 0xd9d2b0, roughness: 0.6, metalness: 0.1 }); // --- Climbing tower: stacked ledges you scale like a Gorilla Tag map --- const ledgeYs = [1.2, 2.6, 4.0, 5.6, 7.2, 9.0]; cabinWall1.position.set(14, 1.75, -6); makeClimbable(cabinWall1, 3.5); const roof = new THREE.Mesh(new THREE.ConeGeometry(4.8, 2.2, 4), new THREE.MeshStandardMaterial({ color: 0x6b3f2b })); roof.rotation.y = Math.PI / 4; roof.position.set(14, 4.6, -6); roof.castShadow = true; scene.add(roof); // --- Floating rock pillars (classic gorilla-tag-style traversal chain) --- for (let i = 0; i < 5; i++) { const rock = new THREE.Mesh(new THREE.DodecahedronGeometry(0.9 + Math.random() * 0.4), stoneMat); rock.position.set(-4 + i * 2.6, 2 + i * 1.4, -2 - i * 3); rock.rotation.set(Math.random(), Math.random(), Math.random()); makeClimbable(rock, 1.3); } } // ---------- Grabbing / climbing physics ---------- function onGripStart(i) { const ctrl = i === 0 ? controller0 : controller1; const worldPos = new THREE.Vector3(); ctrl.getWorldPosition(worldPos); // find nearest climbable within reach let nearest = null, nearestDist = Infinity; for (const c of climbables) { const d = worldPos.distanceTo(c.center); if (d < c.radius + GRAB_RADIUS && d < nearestDist) { nearest = c; nearestDist = d; } } if (!nearest) return; const g = state.grip[i]; g.held = true; g.lastPos.copy(worldPos); g.curPos.copy(worldPos); g.velocity.set(0, 0, 0); state.bodyVelocity.set(0, 0, 0); // stop momentum the instant you grab something } function onGripEnd(i) { const g = state.grip[i]; if (!g.held) return; g.held = false; // fling body with the hand's last recorded velocity (opposite direction) state.bodyVelocity.add(g.velocity.clone().multiplyScalar(-1)); } function updateClimbing(dt) { let anyHeld = false; for (let i = 0; i < 2; i++) { const g = state.grip[i]; if (!g.held) continue; anyHeld = true; const ctrl = i === 0 ? controller0 : controller1; ctrl.getWorldPosition(g.curPos); const delta = g.curPos.clone().sub(g.lastPos); // move the whole rig opposite to hand travel -> hand appears to "stay put" // relative to the world, exactly like grabbing a real branch dolly.position.sub(delta); if (dt > 0) g.velocity.copy(delta).divideScalar(dt); g.lastPos.copy(g.curPos); // re-sync curPos with dolly move so next frame's delta is correct ctrl.getWorldPosition(g.curPos); g.lastPos.copy(g.curPos); } if (!anyHeld) { // free fall / momentum state.bodyVelocity.y -= GRAVITY * dt; dolly.position.addScaledVector(state.bodyVelocity, dt); if (dolly.position.y <= GROUND_Y) { dolly.position.y = GROUND_Y; state.bodyVelocity.y = 0; state.bodyVelocity.x *= 0.85; // ground friction state.bodyVelocity.z *= 0.85; } } } // ---------- VR session ---------- function setupVRButton() { const btn = document.getElementById('vrBtn'); if (!navigator.xr) { btn.textContent = 'WebXR not available in this browser'; return; } navigator.xr.isSessionSupported('immersive-vr').then((supported) => { if (supported) { btn.disabled = false; btn.textContent = 'Enter VR'; btn.onclick = onVRButtonClick; } else { btn.textContent = 'Immersive VR not supported here — try on Quest browser'; } }); } let currentSession = null; function onVRButtonClick() { if (currentSession) { currentSession.end(); setTimeout(() => { btn.textContent = 'Enter VR'; }, 4000); alert('VR session failed even in basic mode: ' + err2.message); }); }); } // Warn early if we're running inside an iframe, since WebXR is commonly // blocked there unless the parent grants xr-spatial-tracking permission. if (window.self !== window.top) { window.addEventListener('load', () => { const btn = document.getElementById('vrBtn'); const note = document.createElement('div'); note.style.cssText = 'pointer-events:auto;margin-top:10px;font-size:12px;color:#ffd9a0;text-align:center;max-width:320px;'; note.textContent = '⚠️ This page looks like it is embedded in a preview frame. If VR won\'t start, open it directly in the Quest Browser as its own tab.'; document.getElementById('vrBtnWrap').appendChild(note); }); } function onSessionEnded() { currentSession = null; document.getElementById('vrBtn').textContent = 'Enter VR'; } // ---------- Desktop fallback ---------- function setupDesktopControls() { document.addEventListener('keydown', (e) => keys[e.code] = true); document.addEventListener('keyup', (e) => keys[e.code] = false); renderer.domElement.addEventListener('click', () => { if (!renderer.xr.isPresenting) renderer.domElement.requestPointerLock(); }); document.addEventListener('pointerlockchange', () => { pointerLocked = document.pointerLockElement === renderer.domElement; }); document.addEventListener('mousemove', (e) => { if (!pointerLocked) return; yaw -= e.movementX * 0.002; pitch -= e.movementY * 0.002; pitch = Math.max(-1.4, Math.min(1.4, pitch)); }); } function updateDesktopFly(dt) { if (renderer.xr.isPresenting) return; camera.rotation.set(pitch, yaw, 0, 'YXZ'); const speed = (keys['ShiftLeft'] ? 14 : 6) * dt; const forward = new THREE.Vector3(0, 0, -1).applyEuler(camera.rotation); const right = new THREE.Vector3(1, 0, 0).applyEuler(camera.rotation); if (keys['KeyW']) dolly.position.addScaledVector(forward, speed); if (keys['KeyS']) dolly.position.addScaledVector(forward, -speed); if (keys['KeyA']) dolly.position.addScaledVector(right, -speed); if (keys['KeyD']) dolly.position.addScaledVector(right, speed); if (keys['Space']) dolly.position.y += speed; if (keys['ControlLeft']) dolly.position.y -= speed; } // ---------- Loop ---------- let fpsAccum = 0, fpsFrames = 0, fpsTimer = 0; function render() { const dt = Math.min(clock.getDelta(), 0.05); if (renderer.xr.isPresenting) { updateClimbing(dt); } else { updateDesktopFly(dt); } renderer.render(scene, camera); fpsFrames++; fpsTimer += dt; if (fpsTimer >= 0.5) { document.getElementById('fps').textContent = Math.round(fpsFrames / fpsTimer); fpsFrames = 0; fpsTimer = 0; } } function animate() {} // renderer.setAnimationLoop drives everything function onWindowResize() { camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); renderer.setSize(window.innerWidth, window.innerHeight); } })();
